QLX Template Format (własny format)

Przykład szablonu JSON
{
  "id": "tpl_abc123",
  "name": "Standard Inventory",
  "tags": ["inventory", "items"],
  "target": "universal",        // or "printer:B1"
  "width_mm": 50,               // universal: mm
  "height_mm": 30,              // printer: ignored (uses PrintWidthPx)
  "elements": [
    {
      "type": "text",
      "x": 2, "y": 2, "width": 30,
      "text": "{{name}}",
      "font": "sans-bold", "size": 16,
      "align": "left"
    },
    {
      "type": "text",
      "x": 2, "y": 12,
      "text": "{{location}}",
      "font": "sans", "size": 10,
      "align": "left"
    },
    {
      "type": "line",
      "x1": 2, "y1": 18, "x2": 48, "y2": 18,
      "thickness": 1
    },
    {
      "type": "qr",
      "x": 35, "y": 2, "size": 15,
      "content": "{{qr_url}}"
    },
    {
      "type": "barcode",
      "x": 2, "y": 20, "width": 46, "height": 8,
      "content": "{{id}}"
    },
    {
      "type": "img",
      "x": 40, "y": 20, "width": 8, "height": 8,
      "src": "asset:logo_abc"    // uploaded asset ID
    }
  ]
}
